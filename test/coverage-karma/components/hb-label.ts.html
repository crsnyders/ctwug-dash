import {customElement, bindable} from "aurelia-framework";
import {CustomFieldLabelType} from "../model/enums";
 
@customElement('hb-label')
export class HBLabel {
  @bindable value: string;
  @bindable type: CustomFieldLabelType;
 
  labelClass: string;
  bind() {
 
    switch (this.type) {
      case CustomFieldLabelType.Heading:
        this.labelClass = 'formConstHeading';
        break;
      case CustomFieldLabelType.Normal:
        this.labelClass = 'formConstNormal';
        break;
      case CustomFieldLabelType.FinePrint:
        this.labelClass = 'formConstFinePrint';
        break;
      default:
        this.labelClass = "text";
    }
  }
}
